function submitForm() { const chk = document.getElementById('confirmCheck'); if (!chk.checked) { chk.style.outline = '2px solid var(--red)'; setTimeout(() => chk.style.outline = '', 2000); return; } const btn = document.querySelector('.btn-submit'); const originalText = btn.innerHTML; btn.innerHTML = 'Enviando a ZOHO... ⏳'; btn.disabled = true; // Recolectamos los datos para enviarlos a tu Google Sheets const formData = new URLSearchParams(); formData.append('dni', document.getElementById('dni').value); formData.append('apellido', document.getElementById('apellido').value); formData.append('nombre', document.getElementById('nombre').value); formData.append('fecha_nacimiento', document.getElementById('fechaNac').value); formData.append('domicilio', document.getElementById('domicilio').value); formData.append('localidad', document.getElementById('localidad').value); formData.append('codigo_postal', document.getElementById('cp').value); formData.append('celular', document.getElementById('celular').value); formData.append('mail', document.getElementById('email').value); formData.append('grupo_familiar', vals.grupo === 'si' ? 'Con Grupo' : 'Solo Titular'); formData.append('obra_social', document.getElementById('obraSocialSelect').value || 'No tiene'); formData.append('alergia', getActiveChips('alergiaChips') || 'No presenta'); formData.append('enfermedad_cronica', getActiveChips('cronicaTitularDetalle') || 'No presenta'); formData.append('plan', document.getElementById('plan').value); formData.append('modalidad_pago', vals.pago || 'Efectivo'); formData.append('vigencia', document.getElementById('vigencia').value); formData.append('observaciones', 'ID: ' + solicitudID); // URL de tu Google Script const scriptURL = 'https://script.google.com/macros/s/AKfycbyRD80bHzltVeqp8k17tacqqVcB4E1MOG9da6jxUf8/exec'; fetch(scriptURL, { method: 'POST', mode: 'no-cors', body: formData }) .then(() => { // Al terminar el envío, mostramos la pantalla de éxito que ya tenías diseñada for (let i = 1; i <= 4; i++) document.getElementById('step' + i).style.display = 'none'; document.getElementById('progressFill').style.width = '100%'; document.getElementById('stepTitle').textContent = '¡Enviado!'; document.getElementById('successScreen').classList.add('show'); document.getElementById('successID').textContent = solicitudID; }) .catch(error => { console.error('Error:', error); alert('Hubo un problema al enviar los datos.'); btn.innerHTML = originalText; btn.disabled = false; }); }
